## Find source and headers
FILE(GLOB_RECURSE HEADERS *.h)
FILE(GLOB_RECURSE SOURCES *.c*)

IF(CMAKE_HAS_SORT)
  LIST(SORT HEADERS)
  LIST(SORT SOURCES)
ENDIF(CMAKE_HAS_SORT)

## Build
INCLUDE_DIRECTORIES(
    ${FLTK_INCLUDE_DIR}
	${NEO_CORE_INCLUDE_DIR}
)

# SET(CMAKE_INSTALL_RPATH "./")
ADD_EXECUTABLE(ProjectWizard ${SOURCES} ${HEADERS})

TARGET_LINK_LIBRARIES(ProjectWizard
  ${FLTK_LIBRARIES}
  ${NEO_CORE_LIBRARIES}
)

INSTALL(TARGETS ProjectWizard DESTINATION ${NEO_INSTALL_DIR})
