## Find source and headers
FILE(GLOB_RECURSE HEADERS *.h)
FILE(GLOB_RECURSE SOURCES *.c*)

IF(CMAKE_HAS_SORT)
  LIST(SORT HEADERS)
  LIST(SORT SOURCES)
ENDIF(CMAKE_HAS_SORT)

## Build
INCLUDE_DIRECTORIES(
    ${FLTK_INCLUDE_DIR}
    ${RAKNET_INCLUDE_DIR}
)

# SET(CMAKE_INSTALL_RPATH "./")
ADD_EXECUTABLE(ProfileViewer ${SOURCES} ${HEADERS})

TARGET_LINK_LIBRARIES(ProfileViewer
  ${FLTK_LIBRARIES}
  ${MCORE_LIBRARIES}
  ${RAKNET_LIBRARIES}
)

INSTALL(TARGETS ProfileViewer DESTINATION ${NEO_INSTALL_DIR})
