## Find source and headers
FILE(GLOB_RECURSE HEADERS *.h)
FILE(GLOB_RECURSE SOURCES *.c*)

IF(CMAKE_HAS_SORT)
  LIST(SORT HEADERS)
  LIST(SORT SOURCES)
ENDIF(CMAKE_HAS_SORT)

## Build
INCLUDE_DIRECTORIES(
    ${FLTK_INCLUDE_DIR}
    ${ZLIB_INCLUDE_DIR}
    ${NEO_CORE_INCLUDE_DIR}
    ${LIBZIP_INCLUDE_DIR}
)

ADD_DEFINITIONS(-DDEFAULT_HOST="www.neo-engine.de")

# SET(CMAKE_INSTALL_RPATH "./")
ADD_EXECUTABLE(NeoStore ${SOURCES} ${HEADERS})

TARGET_LINK_LIBRARIES(NeoStore
  ${FLTK_LIBRARIES}
  ${FLTK_DEPENDENCIES}
  fltk_images
  ${NEO_CORE_LIBRARIES}
  ${ZLIB_LIBRARIES}
  ${LIBZIP_LIBRARIES}
  ${WINSOCK_LIBRARIES}
)

INSTALL(TARGETS NeoStore DESTINATION ${NEO_INSTALL_DIR})
